作用域（变量或者函数可以起作用的访问范围）

  • 全局作用域：在script标签中或者一个独立的js文件中，全局变量在任何位置都可以访问
    var num = 1;  // 全局变量
function fun() {
  str = 'abc';  // 全局变量 不推荐 
}
  • 局部作用域：任何函数的内部都有一个局部作用域，在局部作用域中定义的变量为局部变量，只有在定义变量的函数中可以访问
    function fun() {
  var num = 1;  // 局部变量 
}
console.log(num); // num is not defined
  • 块级作用域：ES6之前没有块级作用域
    // 代码块
{
  var num = 1;
}
console.log(num);  // 1 可以访问到

作用域链

var num = 1;  // 全局作用域 --0级作用域链
function fun1() {
  var num = 2;  // fun1局部作用域 --1级作用域链
  function fun2() {
    // fun2局部作用域 --2级作用域链
    console.log(num)  
  }
  fun2()
}
fun1()  // 变量查找顺序 2级作用域链->1级作用域链->0级作用域链

预解析

  • 在没有出现let const之前，会出现一些奇怪的问题

    // 问题一
console.log(num); // undefined
var num = 1;

// 问题二
fun();   // hello
function fun() {
  console.log('hello')
}

  • 预解析过程（当预解析过程中函数和变量名字相同，函数优先）

    • 变量提升：把变量的声明提升到当前作用域的最上面，不包括变量的赋值

      console.log(num); // undefined
var num = 1;

// 上述代码的预解析
var num;
console.log(num);
num = 5;

    • 函数提升：把函数的声明提升到当前作用域的最上面，不包括函数的调用

      fun();   // hello
function fun() {
  console.log('hello')
}

// 上述代码的预解析
function fun() {
  console.log('hello')
}
 fun();

// 注意
function fn() {
  var a = b = c = 9;
  // 上述代码等价于下面的代码
  var a = 9,  // 局部变量
  b = 9,  // 全局变量
  c = 9;  // 全局变量
}

对象

对象的创建方式

  • 对象字面量
    var person = {
  name: 'Kobe',
  age: '40',
  profession: '篮球运动员',
  sports: function() {
    console.log('打篮球');
  }
}
  • 通过new Object()创建
    // 调用构造函数在内存中创建一个空的对象 
var person = new Object();
// 添加成员和方法
person.name = 'Kobe';
person.age = 40;
person.profession = '篮球运动员';
person.sports = function() {
  console.log('打篮球');
}
  • 工厂方法
    function createPerson(name,age,profession) {
  var person = new Object();
  person.name = name;
  person.age = age;
  person.profession = profession;
  person.sports = function() {
    console.log(this.name + '打篮球');
  }
  return person;
}
var person1 = createPerson('Kobe',40,'篮球运动员');
var person2 = createPerson('James',44,'篮球运动员');
  • 自定义构造函数
    // 帕斯卡命名 每个单词的第一个字母大写
function Person(name,age,profession) {
  // this执行当前对象
  this.name = name;
  this.age = age;
  this.profession = profession;
  this.sports = function() {
    console.log(this.name + '打篮球');
  }
}
var person1 = new Person('Kobe',40,'篮球运动员');
var person2 = new Person('James',44,'篮球运动员');

new关键字的执行过程

  • 在内存中创建一个空的对象
  • 让函数中的this指向刚刚创建的对象
  • 执行构造函数，在构造函数中设置属性和方法
  • 返回了当前对象（内部处理）

this指向

  • 函数中的this指向window
    function fun() {
  console.log(this); // window
}
  • 方法中的this指向当前方法所属的对象
    var obj = {
  name: 'Kobe',
  fn: function() {
    console.log(this); // Object-->obj
  }
}
  • 构造函数中的this指向构造函数创建的对象
    function Fun() {
  this.name = 'Kobe';
  console.log(this); 
}

遍历和删除对象的属性

  var obj = {
    name: 'Kobe',
    age: 40,
    sayHi: function() {
      console.log('hello');
    }
  }
  // 遍历 for in --> key 为属性
  for (var key in obj) {
    console.log(key + '=>' + obj[key])
  }
  // 删除属性
  delete obj.name;
  delete obj.sayHi;

内置对象

  • Math对象
    • Math.PI 圆周率
    • Math.random() 生成随机数
    • Math.floor()/Math.ceil() 向下取整/向上取整
    • Math.round() 取整，四舍五入
    • Math.abs() 绝对值
    • Math.max()/Math.min() 最大值/最小值
    • Math.sin()/Math.cos() 正弦/余弦
    • Math.power()/Math.sqrt() 指数幂/平凡根
  • Array对象
