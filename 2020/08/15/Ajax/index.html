Ajax请求

  • 创建对象

    var xhr = new XMLHttpRequest(); // 标准浏览器
var xhr = new ActiveXObject("Microsoft.XMLHTTP"); // IE老版本（IE6）

  • 准备发送

    • 参数一：请求方式（get/post）
    • 参数二：请求地址
    • 参数三：同步或异步标志位 true->异步 默认为true
    • 请求参数：get请求必须拼接在地址上并且要经过编码处理
      var param = 'username=admin';
// get
xhr.open('get', '请求地址?'+encodeURI(param), true);

    // post
    xhr.open(‘get’, ‘请求地址’, true);
    ```

  • 执行发送动作

    // get
xhr.send(null); // get请求需要null参数
// post
xhr.setRequestHeader("Content-Type","application/x-www-form-urlencoded"); // 设置请求头
xhr.send(param); // post请求参数在send中传递 不需要转码 但要设置请求头

  • 指定回调函数

    • xhr.readyState = 0 表示xhr对象初始化完成
    • xhr.readyState = 1 表示已经发送了请求
    • xhr.readyState = 2 表示浏览器已经收到服务器响应数据，但未解析
    • xhr.readyState = 3 数据解析
    • xhr.readyState = 4 数据解析完成，可以使用 
      xhr.onreadystatechange = function() {  
  if (xhr.readyState == 4) {
      if (xhr.status == 200) {}
  }
}

JSON

  • json数据和普通js对象的区别
    • json数据没有变量
    • json形式的数据结尾没有分号
    • json数据中的键必须用双引号包住

Ajax函数封装

function ajax(url,type,param,dataType,callback) {
    // 处理兼容性
    var xhr = null;
    if (window.XMLHttpRequest) {
        xhr = new XMLHttpRequest();
    } else {
        xhr = new ActiveXObject("Microsoft-XMLHTTP");
    }
    if (type == 'get') {
        url += '?' + encodeURI(param);
    }
    var data = null;
    if (type == 'post') {
        data = param;
        xhr.setRequestHeader("Content-Type","application/x-www-form-urlencoded")
    }
    xhr.open(type, url, true);
    xhr.send(null);
    xhr.onreadystatechange = function() {
        if (xhr.readyState == 4) {
            if (xhr.status == 200) {
                var data = xhr.responseText;
                if (dataType == 'json') {
                    data = JSON.parse(data);
                }
                callback(data);
            }
        } 
    }
}

原生模仿jQuery封装ajax

function ajax(obj) {
    var defaults = {
        type: 'get',
        data: {},
        url: '#',
        dataType: 'text',
        async: true,
        success: function(data) {
            console.log(data);
        }
    }
    for (var key in obj) {
        defaults[key] = obj[key];
    }
    var xhr = null;
    if (window.XMLHttpRequest) {
        xhr = new XMLHttpRequest();
    } else {
        xhr = new ActiveXObject('Microsoft.XMLHTTP');
    }

    var param = '';
    for (var key in obj.data) {
        param += key + '=' + obj.data[key] + '&';
    }
    // 去除最后一个多余的&
    param = param.substring(0, param.length - 1);

    if (defaults.type == 'get') {
        // get请求设置中文转码
        defaults.url += '?' + encodeURI(param);
    }
    xhr.open(defaults.type,defaults.url,defaults.async);
    var dat = null;
    if (defaults.type == 'post') {
        dat = param; 
        // post请求设置请求头
        xhr.setRequestHeader("Content-Type","application/x-www-form-urlencoded");
    }
    xhr.send(dat);
    // 同步
    if (!defaults.async) {
        if (defaults.dataType == 'json') {
            return JSON.parse(xhr.responseText);    
        }
        return xhr.responseText;
    }
    // 异步
    xhr.onreadystatechange = function() {
        if (xhr.readState == 4) {
            if (xhr.status == 200) {
                var data = xhr.responseText;
                if (defaults.dataType == 'json') {
                    data = JSON.parse(data);
                }
                defaults.success(data);
            }
        }
    }
}
