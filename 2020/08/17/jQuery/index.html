-jQuery回顾

关于jQuery

  • 是什么：本质上就是一个JavaScript函数库
  • 使用jQuery的原因
    • DOM实现一个简单的功能需要很多的代码
    • DOM中兼容性问题很多
    • DOM中代码的容错性很差
    • DOM中的window.onload只能有一个

jQuery中的顶级对象$

  • DOM中的顶级对象为document 页面中的顶级对象
  • BOM中的顶级对象为window 浏览器中的顶级对象
  • jQuery中的顶级对象为jQuery或者为$

页面的加载事件

  • DOM中的加载事件（只能执行最后一个）
    window.onload = function() {
  console.log('DOM1');
}
window.onload = function() {
  console.log('DOM2');
}
// 最后的结果为DOM2
  • jQuery中的加载事件
    • 写法一（页面所有的内容加载完毕后触发，包括标签、图片、文字内容）
      $(window).load = function() {
  console.log('jQuery1');
}
$(window).load = function() {
  console.log('jQuery2');
}
// 执行结果为 jQuery1 jQuery2
    • 写法二（页面中基本的标签加载完毕后触发，比写法一速度快一些）
      $(document).ready = function() {
  console.log('jQuery1');
}
$(document).ready = function() {
  console.log('jQuery2');
}
// 执行结果为 jQuery1 jQuery2
    • 写法三（页面中基本的标签加载完毕后触发，比写法一速度快一些）
      $(function() {
  console.log('jQuery1');
})
$(function() {
  console.log('jQuery2');
})
// 执行结果为 jQuery1 jQuery2

jQuery和DOM对象的相互转换

  • DOM对象转jQuery对象
    var btn = document.getElementById('btn'); // DOM对象
$(btn).click(function() {}); // jQuery对象
  • jQuery对象转DOM对象
    var btn = $('#btn'); // jQuery对象
btn.get(0); // DOM对象
btn[0]; // DOM对象

jQuery选择器

  • 根据id来选择–>$('#id的值') 一个
  • 根据标签名获取–>$('标签的名字') 多个
  • 根据类样式的名字–>$('.类样式的名字') 多个
  • 层次选择器–>$('ul>li')
  • 索引选择器
    $('ul>li:eq(4)'); // 获取索引为4的li元素
$('ul>li:gt(4)'); // 获取索引大于4的li元素
$('ul>li:lt(4)'); // 获取索引小于4的li元素
  • 关于获取兄弟元素
    // 获取当前元素的下一个兄弟元素
$('#box').next(); 
// 获取当前元素的后面的所有兄弟元素
$('#box').nextAll();
// 获取当前元素的前一个兄弟元素
$('#box').prev();
// 获取当前元素的前面的所有的兄弟元素
$('#box').prevAll();
// 获取当前元素的所有的兄弟元素
$('#box').siblings();

jQuery样式操作

  • 添加和移除类样式

// 移除
$('#btn').click(function() {
  // 移除cls
  // $('#box').removeClass('cls')
  // 移除所有类样式

    var result = $('#box').hasClass('cls'); // true或false

  • 切换元素的类样式方法toggleClass()

  • 设置和获取坐标

    $('#box').offset({
  "top": 200,
  "left": 200

$('#box').offset().left;
$('#box').offset().top;

  • 滚动出去的距离

    $(document).scrollLeft(); // 数字类型
$(document).scrollTop(); // 数字类型

jQuery中的动画

  • 显示和隐藏
    • 数字参数：1000表示1000毫秒
    • 字符串参数：slow normal fast
      $('#btn1').click(function() {
// $('#box').hide(1000)；
$('#box').hide('slow');
})
$('#btn2').click(function() {
// $('#box').show(1000);
$('#box').show('normal');
})
  • 滑入、滑出和切换
    • 数字参数：1000表示1000毫秒
    • 字符串参数：slow normal fast
      $('#btn1').click(function() {
$('#box').slideUp();
})
$('#btn2').click(function() {
$('#box').slideDown();
})
$('#btn3').click(function() {
$('#box').slideToggle();
})
  • 淡入、淡出和切换
    • 数字参数：1000表示1000毫秒
    • 字符串参数：slow normal fast
      $('#btn1').click(function() {
$('#box').fadeIn();
})
$('#btn2').click(function() {
$('#box').fadeOut();
})
$('#btn3').click(function() {
$('#box').fadeToggle();
})
  • 透明度变化
    • 参数一：时间
    • 参数二：透明度
      $('#btn').click(function() {
$('#box').fadeTo(1000, 0.3);
})
  • animate方法
    • 参数一：对象
    • 参数二：时间
    • 参数三：匿名函数
      $('#btn').click(function() {
$('#box').animate({
  "width": "300px",
  "height": "300px"
}, 1000, function() {
  $('#box').animate({
    "width": "100px",
    "height": "100px"
  }, 1000)
})
})
  • stop()停止动画的方法

jQuery动态创建元素

  • $('标签的代码')

    var a = $('<a href="http://www.baidu.com">百度</a>');
$('#box').append(a); // 追加到某个元素后面
$('#box').prepend(a); // 追加到某个元素前面
$('#box').after(a); // 把元素追加到当前元素的后面（兄弟元素）
$('#box').before(a); // 把元素追加到当前元素的前面（兄弟元素）
a.appendTo($('#box')); // 追加到box中

  • 对象.html('标签的代码')

    $('#box').html('<p>p标签</p>');

  • 移除和克隆内容

    $('#box').html(); // 移除box中的元素
$('#box').empty(); // 移除box中的元素
$('#box').remove(); // 移除元素自身

var box = $('#box').clone(); // 克隆box元素

事件绑定和解绑

  • bind() unbind() 不推荐
    // 绑定单个事件
$('#btn').bind('click', function() {
})
// 解绑
$('#btn').unbind('click');
// 绑定多个事件
$('#btn').click().mouseover().mouseout();
$('#btn').bind({
  "click": function() {}
}, {
  "mouseover": function() {}
},{
  "mouseout": function() {} 
})
  • delegate() undelegate() 不推荐
    // 为box中的p标签绑定点击事件
$('#box').delegate('p','click',function() {
})
// 解绑
$('#box').undelegate('p','click');
  • on() off()
    // 给按钮绑定点击事件
$('#btn').on('click', function() {
})
// 解绑
$('#btn').off('click');
// 给box中的p标签绑定点击事件
$('#box').on('click','p',function() {
})

触发事件

  • 方式一
    $('#btn').click()
  • 方式二
    $('#btn').trigger('click');
  • 方式三
    $('#btn').triggerHandler('click');

事件对象

$('#box').on('click','input',function(event) {
  /**
  * event.delegateTarget-->指向的是box
  * event.currentTarget-->指向的是input
  * event.target-->指向的是input
  */ 
})

事件冒泡及取消事件冒泡

// box1中有box2，box2中有box3
$('#box1').click(function() {
  console.log('box1');
})
$('#box2').click(function() {
  console.log('box2');
})
$('#box3').click(function() {
  console.log('box3');
  return false; // 取消事件冒泡
})
// 点击box3 结果打印box3 box2 box1（事件冒泡）
