面向对象

构造函数的原型（通过原型创建的方法，所有构造函数的实例对象都可以访问）

function Person(name, age) {
    this.name = name;
    this.age = age;
}
Person.prototype.getName = function() {
    return this.name;
}
var person = new Person('Kobe', 40);
console.log(person.getName()) // Kobe

对象的原型

function Person(name, age) {
    this.name = name;
    this.age = age;
    this.getName = function() {
        return this.name;
    }
}
var person = new Person('Kobe', 40);
console.log(person.__proto__ === Person.prototype); // true
  • 当调用对象的属性或方法时，先去找对象本身的属性或方法，如果对象没有该属性或方法，此时则调用原型中的属性或方法
  • 当设置对象的属性时，不会查询原型链，如果对象没有该属性，直接添加该属性

自调用函数（解决命名冲突，连续写多个自调用函数时要加分号）

;(function() {
    // 局部作用域 外部访问不到
    var src = 'abc';
    window.src = src; // 外部能访问到
})()
  • 自调用函数的参数
    ;(function(w, u) {
    // 传入window让变量名可以被压缩
    // 老版浏览器传入undefined防止值被改变
})(window, undefined)

改变this指向

  • bind
    var src = 'Kobe';
function fun1() {
    console.log(this.src)
}
fun1(); // Kobe
var o = {src: 'James'}
var fun2 = fun1.bind(o); // bind第一个参数改变this指向
fun2(); // James

对象之间的拷贝

function extend(parent, child) {
    for (var key in parent) {
        if (child[key]) {
            continue;
        }
        child[key] = parent[key];
    }
}

Student.prototype = new Person();
Student.prototype.constructor = Student;

bind call apply

  • bind改变this的指向，复制一个新的函数（不调用函数）
    function sum(x, y) {
    console.log(this);
    return x + y;
}
sum(1,3); // this->window 相当于window.sum(1,3);
var obj = {name: 'Kobe'};
var fun = sum.bind(obj, 1, 3);
fun(); // this->obj
// 应用
var o = {
    name: 'James',
    fn: function() {
        setInterval(function() {
            console.log(this.name); // 没调用bind前this指向window,调用bind后指向o
        }.bind(this), 1000)
    }
}
  • call改变this的指向，直接调用函数，返回值就是函数的返回值
    function sum(x, y) {
    console.log(this);
    return x + y;
}
var obj = {name: 'Kobe'};
sum.call(obj, 1, 3); // this->obj
  • apply改变this的指向，直接调用函数，返回值就是函数的返回值，第二个参数是一个数组，把数组的每一项展开传给函数
    var arr = [2, 4, 6, 8, 0, 1];
console.log(Math.max(arr)); // NaN
// 应用
console.log(Math.max.apply(Math, arr)); // 8
console.log.apply(console, arr); // 2 4 6 8 0 1

借用构造函数实现继承

function Person(name, age) {
    this.name = name;
    this.age = age;
}
function Student(name,age,score) {
    Person.call(this, name, age); // 借用构造函数以及call
    this.score = score;
}

组合继承（原型继承+组合继承）

function Person(name, age) {
    this.name = name;
    this.age = age;
}
Person.prototype.sayHi = function () {
    console.log('hello' + this.name);
}
function Student(name, age, score) {
    Person.call(this, name, age); // 借用构造函数
    this.score = score;
}
Student.prototype = new Person(); // 让子类型继承父类型的方法 这一步可以给子类型添加独有的方法而不会影响父类型
Student.prototype.constructor = Student;

函数进阶

函数的定义方式

  • 函数声明

    function fun() {
    console.log('hello');
}

  • 函数表达式

    var fun = function() {
    console.log('hello');
}

  • new Function() 代码解析较慢

    var fun = new Function('a','b',console.log(a + b)')

  • 函数声明和表达式的区别

    fun1();   // fun1
function fun1() {
    console.log('fun1');
}
fun2(); // 报错
var fun2 = function() {
    console.log('fun2');
}

// 以上分析（预解析过程）
function fun1() {
    console.log('fun1');
}
var fun2;
fun1(); // fun1
fun2(); // 此时不是函数
fun2 = function() {
    console.log('fun2');
}

函数调用方式以及内部this指向

  • 普通函数调用
    function fun() {
    console.log(this); // window
}
fun();
  • 作为方法调用
    var obj = {
    fun: function() {
        console.log(this); // obj
    }
}
obj.fun();
  • 作为构造函数调用 内部this指向由该构造函数创建的对象
  • 作为事件的处理函数 this指向事件源
  • 作为定时器的参数
    setTimeout(function() {
    console.log(this); // window
}, 1000)

闭包

  • 概念：闭包是使用被作用域封闭的变量，函数，闭包等执行的一个函数的作用域，通常用和其相应的函数来指代这些作用域（可以独立访问数据的函数），个人理解就是一个作用域中访问到另一个作用域中的变量或函数的一种现象
  • 闭包的好处：延展了作用域的范围

拷贝

浅拷贝

var obj1 = {
    name: 'Kobe',
    age: 40,
    friend: {
        name: 'James',
        age: 44
    }
}
var obj2 = {}
for (var key in obj1) {
    obj2[key] = obj1[key];  // 复杂类型复制的地址
}
// 浅拷贝 修改ojb1中的name、age不会影响obj2，但修改obj1中friend中的name、age会影响obj2中friend

深拷贝

var obj1 = {
    name: 'Kobe',
    age: 40,
    friend: {
      name: 'James',
      age: 44
    },
    hobbies: ['看电影', '打篮球']
};
// obj1拷贝给obj2
function deepCopy(obj1, obj2) {
    for (var key in obj1) {
      var item = obj1[key];
      if (item instanceof Array) {
        obj2[key] = [];
        deepCopy(item, obj2[key])
      } else if (item instanceof Object) {
        obj2[key] = {};
        deepCopy(item, obj2[key])
      } else {
        obj2[key] = item;
      }
    }
}
var obj2 = {}
deepCopy(obj1, obj2);
obj1.friend.name = 'YaoMing';
console.log(obj2.friend.name); // James

正则表达式（匹配规律规则的表达式）

内置对象RegExp

var regularExpression = new RegExp('ab[a-z]', 'i');
var str = 'aBc';
console.log(regularExpression.test(str)); // true
  • 第一个参数 模式pattern
  • 第二个参数 flag i->忽略大小写 g->全局匹配

简单写法

var regularExpression = /ab[a-z]/i;
var str = 'aBc';
console.log(regularExpression.test(str)); // true

正则方法

  • RegExp对象的方法

    • test()
      var regExp = /ab[a-z]/i;
var str = 'aBc';
console.log(regExp.test(str)); // true
    • exec()无论调用多少次只返回一个匹配到的结果，返回所有结果就需要全局匹配并多次调用
      var str = 'Kobe:44,James: 40';
var regExp = /\d+/g;
do {
    var result = regExp.exec(str); // 数组
    if (result != null) {
        console.log(result[0])
    }
} while(result != null)

  • String对象的方法

    • split()

    • match()

      var str = 'Kobe:44,James: 40';
var regExp = /\d+/g; // 需要全局匹配
console.log(str.match(regExp)); // 数组

// 分组提取
var str = '2013-4-3';
var regExp = /(\d{4})-(\d{1,2})-(\d{1,2})/; // 分组提取
 str.match(regExp);
console.log(RegExp.$1); // 2013
console.log(RegExp.$2); // 4
console.log(RegExp.$3); // 3

    • replace()

      var str = ' aaa   dd dfsafa  d  ';
// 去掉空格
str.replace(/\s/g, '');
